pipeline {
    agent any
    
    stages {
        stage('Checkout') {
            steps {
                // Checkout code from the Git repository
                git url: 'https://github.com/SujalMohite-CodeMaster/Jenkins-Pipeline.git'
            }
        }

        stage('Build') {
            steps {
                // Build the project using Maven
                sh 'mvn clean install'
            }
        }

        stage('Test') {
            steps {
                // Run unit tests
                sh 'mvn test'
            }
        }

        stage('Package') {
            steps {
                // Package the application (optional)
                sh 'mvn package'
            }
        }
    }

    post {
        always {
            // Cleanup or notifications (optional)
            echo 'Pipeline completed.'
        }
    }
}
